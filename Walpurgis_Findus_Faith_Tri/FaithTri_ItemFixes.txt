// Patch TorchW to grant PowerTorch so ACS scripts in Faith Tri function correctly.
// This allows the Walpurgis light effect and the Faith inheritance logic to coexist.

// This invisible actor waits 15 seconds then removes the PowerTorch
ACTOR PowerTorchLimiter
{
    +NOBLOCKMAP
    +NOGRAVITY
    +DONTSPLASH
    -COUNTITEM
    States
    {
    Spawn:
        TNT1 A 0
        // 12 seconds = 420 tics (35 tics * 12)
        TNT1 A 420
        // Remove the hardcoded light effect from the player
        TNT1 A 0 A_TakeInventory("PowerTorch", 1, 0, AAPTR_MASTER)
        Stop
    }
}

ACTOR TorchW_FaithPatch : TorchW replaces TorchW
{
    States
    {
    Use:
        TNT1 A 0 A_SpawnItemEx("TorchyLight", 0, 0, 0, 0, 0, 0, 0, SXF_SETMASTER)
        TNT1 A 0 A_GiveInventory("PowerTorch")
        TNT1 A 0 A_SpawnItemEx("PowerTorchLimiter", 0, 0, 0, 0, 0, 0, 0, SXF_SETMASTER)
        Stop
    }
}
