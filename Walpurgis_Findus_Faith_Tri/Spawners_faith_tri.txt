// --- LIGHTBRINGER ---
// Weapon: Lightbringer (Slot 7)

ACTOR BlueWeaponFaith : BlueWeapon replaces BlueWeapon
{
	States
	{
	LBSpawn:
		TNT1 A 0 A_JumpIfInventory("WalLightbringer", 1, "OnlyMana",AAPTR_PLAYER1)
		TNT1 A 0 A_SpawnItemEx("WalLightbringer",0,0,0,0,0,0,0,SXF_TRANSFERPOINTERS|SXF_TRANSFERAMBUSHFLAG|SXF_TRANSFERSPECIAL|SXF_CLEARCALLERTID|SXF_CLEARCALLERSPECIAL|SXF_SETMASTER,0,tid)
		TNT1 A 0 A_JumpIf((multiplayer && !deathmatch),"SpawnMulti")
		Goto EndProcess
	CrusIISpawn:
		TNT1 A 0 A_JumpIfInventory("WalLightbringer", 1, "SpearSpawn",AAPTR_PLAYER1)
		TNT1 A 0 A_JumpIfInventory("Sturmspear", 1, "LBSpawn",AAPTR_PLAYER1)
		TNT1 A 0 {
			if (random(1,2) == 1) // Randomly pick if player has neither.
			{
				A_SpawnItemEx("Sturmspear",0,0,0,0,0,0,0,SXF_TRANSFERPOINTERS|SXF_TRANSFERAMBUSHFLAG|SXF_TRANSFERSPECIAL|SXF_CLEARCALLERTID|SXF_CLEARCALLERSPECIAL|SXF_SETMASTER,0,tid);
			}
			else 
			{
				A_SpawnItemEx("WalLightbringer",0,0,0,0,0,0,0,SXF_TRANSFERPOINTERS|SXF_TRANSFERAMBUSHFLAG|SXF_TRANSFERSPECIAL|SXF_CLEARCALLERTID|SXF_CLEARCALLERSPECIAL|SXF_SETMASTER,0,tid);
			}}
		TNT1 A 0 A_JumpIf((multiplayer && !deathmatch),"SpawnMulti")
		Goto EndProcess
	CrusSpecSpawn:
		TNT1 A 0 A_JumpIfInventory("WalLightbringer", 1, "OnlyMana",AAPTR_PLAYER1)
		TNT1 A 0 A_JumpIfInventory("Sturmspear", 1, "OnlyMana",AAPTR_PLAYER1)
		TNT1 A 0 {
			if (random(1,2) == 1) // Randomly pick if player has neither.
			{
				A_SpawnItemEx("WalLightbringer",0,0,0,0,0,0,0,SXF_TRANSFERPOINTERS|SXF_TRANSFERAMBUSHFLAG|SXF_TRANSFERSPECIAL|SXF_CLEARCALLERTID|SXF_CLEARCALLERSPECIAL|SXF_SETMASTER,0,tid);
				A_GiveInventory("SpecialArmamentsChosenBlue",1,AAPTR_PLAYER1); 
			}
			else 
			{
				A_SpawnItemEx("Sturmspear",0,0,0,0,0,0,0,SXF_TRANSFERPOINTERS|SXF_TRANSFERAMBUSHFLAG|SXF_TRANSFERSPECIAL|SXF_CLEARCALLERTID|SXF_CLEARCALLERSPECIAL|SXF_SETMASTER,0,tid);
				A_GiveInventory("SpecialArmamentsChosenBlue",1,AAPTR_PLAYER1); 
			}}
		TNT1 A 0 A_JumpIf((multiplayer && !deathmatch),"SpawnMulti") 
		Goto EndProcess
	}
}
